2) Verificar se uma pessoa pode entrar em uma festa
Condição1: A pessoa pode entrar se for maior de idade OU se estiver 
acompanhada de um responsável.

algoritmo "Entrada na festa"

var
  idade: inteiro
  acompanhado: logico
  pode_entrar: logico

inicio
  escreva("Digite a idade: ")
  leia(idade)

  escreva("Está acompanhado? (verdadeiro/falso): ")
  leia(acompanhado)

  se (idade >= 18) OU acompanhado entao
     escreva("Entrada permitida.")
  senao
     escreva("Entrada negada.")
  fimse
