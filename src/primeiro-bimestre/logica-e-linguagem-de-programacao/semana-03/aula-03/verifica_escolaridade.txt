algoritmo "verifica_escolaridade"

var
  idade: inteiro
  media: real 

inicio
  escreva("Digite uma idade: ")
  leia(idade)
  escreva(" Digite uma média: ")
  leia(media)
    se (idade >= 11) entao
      se (idade <= 13) entao
        escreva(" Aluno: Ensino Fundamental 2")
      senao
        se (idade <= 17) entao
          escreva(" Aluno: Ensino Médio")
        senao
          escreva(" Aluno: EJA")
        fimse
      fimse
    senao
      escreva(" Aluno não pode ser classificado!")
    fimse
    se (media >= 0) entao
      se (media <= 10) entao
        se (media >= 8.5) entao
          escreva(" Desenpenho de Excelência")
        senao
          se (media >= 7.0) entao
          escreva(" Desenpenho Satisfatório")
          senao
            se (media >= 5.0) entao
              escreva(" Desenpenho de Recuperação")
            senao
              escreva(" Desempenho de Reforço intensivo")
            fimse
          fimse
        fimse
      senao
        escreva(" A Média  maior que 10 não pode ser classificada")
      fimse
    senao
      escreva(" A Média menor que 0 não pode ser classificada")
    fimse     
fimalgoritmo
